בתרגיל ג'אווה סקריפט הבא נלמד להדפיס בצורה מותנה באמצעות פונקציה שמקבלת פרמטר ומדפיסה איבר, כמספר הפעמים הנדרש

על מנת לראות הודעות שחוזרות על עצמן יש לבטל את הוי על יד האופציה group similar messages in console.

  • a. prompt the user to enter two numbers one for the number of rows and 1 for the number of items in each row.
  • b. generate a block of Asterisks and print them.
  • c. bonus: allow the user to choose the symbol he like to print. 

let numOfRows = +prompt('how many rows?')
let numOfSymbols = +prompt('how many items per row?')
let symbol = '*'

const init = () => {
    for (let i = 0; i < numOfRows; i++) {
        printAsterisks(numOfSymbols)
    }
}

const printAsterisks = (numOfPrints) => {
    let currChar = `${symbol}`
    for (let i = 1; i < numOfPrints; i++) {
        currChar += symbol
    }
    console.log(currChar)
}
