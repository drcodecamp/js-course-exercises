בתרגיל ג'אווה סקריפט הבא נלמד איך להגדיר מצבים שונים שדורשים אותו טיפול בקוד במקרים מסויימים, אף נלמד איך להתמודד עם מצב שבו אנחנו מקבלים ערך לא ידוע וכל זאת בשימוש תנאי switch.

  • a. consider this Tree of colors
    Red Tree: red, salmon, tomato, crimson
    Green Tree: green, lime, teal
    Yellow Tree: gold, yellow, khaki
  • b. prompt the user to enter one of his favorite color.
  • c. print the name of the group related to his selection, if the user enters "red", print, "red is part of the red tree"
  • d. Bonus: if the user didn't enter any color that we can recognize, print "is that a new color"? 

var color = prompt('Please enter your fav color!')
switch (color) {
    case 'red':
    case 'salmon':
    case 'tomato':
    case 'crimson':
        console.log(color, 'Belong to Red Tree.')
        break
    case 'green':
    case 'lime':
    case 'teal':
        console.log(color, 'Belong to Green Tree.')
        break
    case 'gold':
    case 'yellow':
    case 'khaki':
        console.log(color, 'Belong to Yellow Tree.')
        break
    default:
        console.log('is that a new color?')
}
